Документация по PublicationApi

PublicationApi работает несложно, начнём разбирать базовые функции для простых пользователей. Первое, что мы сделаем - создадим публикацию:

from requests import post

link_website = 'https://rjkzavrs-studio.herokuapp.com/'

# Для работы требуется email и password User
print(post(f'{link_website}api/publication_create/test_account@yandex.ru/test_123',
                   json={'id': 10,
                         'header': 'Тестовая запись',
                         'body': 'Тестовая запись с тестовой информацией.'
                                 'Тестовая информация для тестовой записи'}).json())

# При выполнении должен вернуть {'success': 'Publication Тестовая запись created'}}

Теперь давайте получим о ней информацию:

from requests import post, get

link_website = 'https://rjkzavrs-studio.herokuapp.com/'

# Для работы требуется email и password User
print(get(f'{link_website}api/developers_diary/test_account@yandex.ru/test_123/10').json())


# При выполнении должен вернуть {'publication': {'bad_marks': 0, 'body':
        'Тестовая запись с тестовой информацией.Тестовая информация для тестовой записи',
        'comments': [], 'created_date': '2020-04-19 15:42', 'good_marks': 0, 'header':
        'Тестовая запись', 'id': 10}}

Теперь изменим информацию публикации:

from requests import post, get, put

link_website = 'https://rjkzavrs-studio.herokuapp.com/'

# Для работы требуется email и password User
print(put(f'{link_website}api/publication/test_account@yandex.ru/test_123/10',
                  json={'header': 'Тестовая смена загаловка'}).json())

# При выполнении должен вернуть {'success': 'Publication Тестовая смена загаловка changed'}

Ну, и удалим:

from requests import post, get, put, delete

link_website = 'https://rjkzavrs-studio.herokuapp.com/'

print(delete(f'{link_website_2}api/publication/test_account@yandex.ru/test_123/10').json())
